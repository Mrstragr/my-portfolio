Code Example: Sudoku Solver

bool solveSudoku(int grid[N][N]) {
  int row, col;

  // Find empty cell
  if (!findEmptyCell(grid, &row, &col))
    return true; // Solved!

  // Try digits 1-9
  for (int num = 1; num <= 9; num++) {
    if (isSafe(grid, row, col, num)) {
      grid[row][col] = num;

      // Recursively solve
      if (solveSudoku(grid))
        return true;

      // Backtrack if fails
      grid[row][col] = 0;
    }
  }
  return false; // Triggers backtracking
}
